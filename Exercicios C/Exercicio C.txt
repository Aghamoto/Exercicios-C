for (i = 0; i<10; i++)
{
printf("Digite um numero:");
scanf("%d",&x[i]);
}

for (i=0;i<10; i++)
for(j=i+1;j<10;j++)
{
if (x[i]>x[j])
{
temp=x[i];
x[i]=x[j];
x[j]=temp;
}
}

printf("Vetor em ordem crescente: \n");
for(i=0;i<10;i++)
printf(" %d",x[i]); 

system("pause");
return 0;
}
